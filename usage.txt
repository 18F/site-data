Python 2 only (for now)

set environment variables:
  export GITHUB_USER=
  export GITHUB_AUTH=
  # PROD and STAGING are urls for `python manage.py rebuild`
  export PROD=
  export STAGING=
  # port to serve from
  export VCAP_APP_PORT=8080
  export ENV=local
  # choose your own
  export HTUSER=catherine
  export HTAUTH=mnhatsoff2thee
  export FLASK_CONFIG=development

running under Vagrant requires host 0.0.0.0
python manage.py runserver -h 0.0.0.0

python manage.py db migrate # generates new migrations
python manage.py db upgrade # applies the migrations
